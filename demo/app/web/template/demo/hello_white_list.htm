class helloController extends Controller {

	public $initphp_list = array("white_list"); //Action白名单
	
	/**
	 * 白名单使用 white_list方法名称需要放置到$initphp_list这个变量中
	 */
	public function white_list() {
		echo "<br/><h1>白名单使用</h1>";
		$this->view->display("demo/hello_white_list"); //使用模板
	}
}
